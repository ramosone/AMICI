38 void BackwardProblem::workBackwardProblem() {
39 
40 
41  if (model->nx_solver <= 0 || solver->getSensitivityOrder() < SensitivityOrder::first ||
42  solver->getSensitivityMethod() != SensitivityMethod::adjoint || model->nplist() == 0) {
43  return;
44  }
45 
46  solver->setupB(this, model);
47 
48  int it = rdata->nt - 2;
49  --iroot;
50 
51  while (it >= 0 || iroot >= 0) {
52 
53  /* check if next timepoint is a discontinuity or a data-point */
54  double tnext = getTnext(discs, iroot, it);
55 
56  if (tnext < t) {
57  solver->solveB(tnext, AMICI_NORMAL);
58  solver->getB(which, &t, &xB, &dxB);
59  solver->getQuadB(which, &t, &xQB);
60  solver->getDiagnosisB(it, rdata, this->which);
61  }
62 
63  /* handle discontinuity */
64  if (model->ne > 0 && rdata->nmaxevent > 0 && iroot >= 0) {
65  if (tnext == discs.at(iroot)) {
66  handleEventB(iroot);
67  --iroot;
68  }
69  }
70 
71  /* handle data-point */
72  if (tnext == rdata->ts[it]) {
73  handleDataPointB(it);
74  it--;
75  }
76 
77  /* reinit states */
78  solver->reInitB(which, t, &xB, &dxB);
79  solver->quadReInitB(which, &xQB);
80  solver->calcICB(which, t, &xB, &dxB);
81  }
82 
83  /* we still need to integrate from first datapoint to tstart */
84  if (t > model->t0()) {
85  /* solve for backward problems */
86  solver->solveB(model->t0(), AMICI_NORMAL);
87  solver->getQuadB(which, &(t), &xQB);
88  solver->getB(which, &(t), &xB, &dxB);
89  solver->getDiagnosisB(0, rdata, this->which);
90  }
91 
92  computeLikelihoodSensitivities();
93 }

96 void BackwardProblem::handleEventB(const int iroot) {
97  for (int ie = 0; ie < model->ne; ie++) {
98 
99  if (rootidx[iroot * model->ne + ie] == 0) {
100  continue;
101  }
102 
103  model->fdeltaqB(ie, t, &x_disc[iroot],&xB,&xdot_disc[iroot], &xdot_old_disc[iroot]);
104  model->fdeltaxB(ie, t, &x_disc[iroot],&xB,&xdot_disc[iroot], &xdot_old_disc[iroot]);
105 
106  for (int ix = 0; ix < model->nxtrue_solver; ++ix) {
107  for (int iJ = 0; iJ < model->nJ; ++iJ) {
108  xB[ix + iJ * model->nxtrue_solver] +=
109  model->deltaxB[ix + iJ * model->nxtrue_solver];
110  if (model->nz > 0) {
111  xB[ix + iJ * model->nxtrue_solver] +=
112  dJzdx[iJ + ( ix + nroots[ie] * model->nx_solver ) * model->nJ];
113  }
114  }
115  }
116 
117  for (int iJ = 0; iJ < model->nJ; ++iJ) {
118  for (int ip = 0; ip < model->nplist(); ++ip) {
119  xQB[ip + iJ * model->nplist()] +=
120  model->deltaqB[ip + iJ * model->nplist()];
121  }
122  }
123 
124  nroots[ie]--;
125  }
126 
127  model->updateHeavisideB(&rootidx[iroot * model->ne]);
128 }

152 void BackwardProblem::computeLikelihoodSensitivities()
153 {
154  for (int iJ = 0; iJ < model->nJ; iJ++) {
155  if (iJ == 0) {
156  for (int ip = 0; ip < model->nplist(); ++ip) {
157  llhS0[ip] = 0.0;
158  for (int ix = 0; ix < model->nxtrue_solver; ++ix) {
159  llhS0[ip] += xB[ix] * sx.at(ix,ip);
160  }
161  }
162  } else {
163  for (int ip = 0; ip < model->nplist(); ++ip) {
164  llhS0[ip + iJ * model->nplist()] = 0.0;
165  for (int ix = 0; ix < model->nxtrue_solver; ++ix) {
166  llhS0[ip + iJ * model->nplist()] +=
167  xB[ix + iJ * model->nxtrue_solver] * sx.at(ix,ip)+
168  xB[ix] * sx.at(ix + iJ * model->nxtrue_solver,ip);
169  }
170  }
171  }
172  }
173 
174  for (int iJ = 0; iJ < model->nJ; iJ++) {
175  for (int ip = 0; ip < model->nplist(); ip++) {
176  if (iJ == 0) {
177  rdata->sllh.at(ip) -= llhS0[ip] + xQB[ip*model->nJ];
178  } else {
179  rdata->s2llh.at(iJ - 1 + ip * (model->nJ - 1)) -=
180  llhS0[ip + iJ * model->nplist()] +
181  xQB[iJ + ip*model->nJ];
182  }
183  }
184  }
