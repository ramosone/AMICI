156 void ReturnData::applyChainRuleFactorToSimulationResults(const Model *model) {
157  // chain-rule factor: multiplier for am_p
158  std::vector<realtype> coefficient(nplist, 1.0);
159  std::vector<realtype> pcoefficient(nplist, 1.0);
160 
161  std::vector<realtype> unscaledParameters = model->getParameters();
162  unscaleParameters(unscaledParameters, model->getParameterScale(), unscaledParameters);
163 
164  std::vector<realtype> augcoefficient(np, 1.0);
165 
166  if (sensi == SensitivityOrder::second && o2mode == SecondOrderMode::full) {
167  for (int ip = 0; ip < np; ++ip) {
168  switch (pscale[ip]) {
169  case ParameterScaling::log10:
170  augcoefficient.at(ip) = unscaledParameters.at(ip) * log(10);
171  break;
172  case ParameterScaling::ln:
173  augcoefficient.at(ip) = unscaledParameters.at(ip);
174  break;
175  case ParameterScaling::none:
176  break;
177  }
178  }
179  }
180 
181  for (int ip = 0; ip < nplist; ++ip) {
182  switch (pscale[model->plist(ip)]) {
183  case ParameterScaling::log10:
184  coefficient.at(ip) = log(10.0);
185  pcoefficient.at(ip) = unscaledParameters.at(model->plist(ip)) * log(10);
186  break;
187  case ParameterScaling::ln:
188  pcoefficient.at(ip) = unscaledParameters.at(model->plist(ip));
189  break;
190  case ParameterScaling::none:
191  break;
192  }
193  }
194 
195  if (sensi >= SensitivityOrder::first) {
196  // recover first order sensitivies from states for adjoint sensitivity
197  // analysis
198  if (sensi == SensitivityOrder::second && o2mode == SecondOrderMode::full) {
199  if (sensi_meth == SensitivityMethod::adjoint) {
200  for (int ip = 0; ip < nplist; ++ip)
201  for (int ix = 0; ix < nxtrue; ++ix)
202  for (int it = 0; it < nt; ++it)
203  sx.at(ix + nxtrue*(ip + it*nplist)) =
204  x.at(it * nx + nxtrue + ip * nxtrue + ix);
205 
206  for (int ip = 0; ip < nplist; ++ip)
207  for (int iy = 0; iy < nytrue; ++iy)
208  for (int it = 0; it < nt; ++it)
209  sy.at(iy + nytrue*(ip + it*nplist)) =
210  y.at(it * ny + nytrue + ip * nytrue + iy);
211 
212  for (int ip = 0; ip < nplist; ++ip)
213  for (int iz = 0; iz < nztrue; ++iz)
214  for (int it = 0; it < nt; ++it)
215  sz.at(iz + nztrue*(ip + it*nplist)) =
216  z.at(it * nz + nztrue + ip * nztrue + iz);
217  }
218  }
219 
220  for (int ip = 0; ip < nplist; ++ip)
221  sllh.at(ip) *= pcoefficient.at(ip);
222 
223  if(!sres.empty())
224  for (int iyt = 0; iyt < nytrue*nt; ++iyt)
225  for (int ip = 0; ip < nplist; ++ip)
226  sres.at((iyt * nplist + ip)) *= pcoefficient.at(ip);
227 
228  if(!FIM.empty())
229  for (int ip = 0; ip < nplist; ++ip)
230  for (int jp = 0; jp < nplist; ++jp)
231  FIM.at(jp + ip * nplist) *= pcoefficient.at(ip)*pcoefficient.at(jp);
232 
233 #define chainRule(QUANT, IND1, N1T, N1, IND2, N2) \
234  if (!s##QUANT.empty()) \
235  for (int IND1 = 0; IND1 < N1T; ++IND1) \
236  for (int ip = 0; ip < nplist; ++ip) \
237  for (int IND2 = 0; IND2 < N2; ++IND2) { \
238  s##QUANT.at((IND2*nplist + ip)*N1 + IND1) *= \
239  pcoefficient.at(ip); \
240  }
241 
242  chainRule(x, ix, nxtrue, nx, it, nt);
243  chainRule(y, iy, nytrue, ny, it, nt);
244  chainRule(sigmay, iy, nytrue, ny, it, nt);
245  chainRule(z, iz, nztrue, nz, ie, nmaxevent);
246  chainRule(sigmaz, iz, nztrue, nz, ie, nmaxevent);
247  chainRule(rz, iz, nztrue, nz, ie, nmaxevent);
248  chainRule(x0, ix, nxtrue, nx, it, 1);
249  }
250 
251  if (o2mode == SecondOrderMode::full) { // full
252  if (!s2llh.empty() && !sllh.empty()) {
253  for (int ip = 0; ip < nplist; ++ip) {
254  for (int iJ = 1; iJ < nJ; ++iJ) {
255  s2llh[ip * nplist + (iJ - 1)] *=
256  pcoefficient.at(ip) * augcoefficient[iJ - 1];
257  if (model->plist(ip) == iJ - 1)
258  s2llh[ip * nplist + (iJ - 1)] +=
259  sllh.at(ip) * coefficient.at(ip);
260  }
261  }
262  }

264 #define s2ChainRule(QUANT, IND1, N1T, N1, IND2, N2) \
265  if (!s##QUANT.empty()) \
266  for (int ip = 0; ip < nplist; ++ip) \
267  for (int iJ = 1; iJ < nJ; ++iJ) \
268  for (int IND1 = 0; IND1 < N1T; ++IND1) \
269  for (int IND2 = 0; IND2 < N2; ++IND2) { \
270  s##QUANT.at((IND2*nplist + ip)*N1 + IND1 + iJ*N1T) *= \
271  pcoefficient.at(ip) * augcoefficient[iJ - 1]; \
272  if (model->plist(ip) == iJ - 1) \
273  s##QUANT.at((IND2*nplist + ip)*N1 + IND1 + iJ*N1T) += \
274  s##QUANT.at((IND2*nplist + ip)*N1 + IND1) * \
275  coefficient[ip]; \
276 }

285  }

288  for (int ip = 0; ip < nplist; ++ip) {
289  s2llh.at(ip) *= pcoefficient.at(ip);
290  s2llh.at(ip) += model->k()[nk - nplist + ip] * sllh.at(ip) /
291  unscaledParameters[model->plist(ip)];
292  }
293 
294 #define s2vecChainRule(QUANT, IND1, N1T, N1, IND2, N2) \
295  if (!s##QUANT.empty()) \
296  for (int ip = 0; ip < nplist; ++ip) \
297  for (int IND1 = 0; IND1 < N1T; ++IND1) \
298  for (int IND2 = 0; IND2 < N2; ++IND2) { \
299  s##QUANT.at((IND2*nplist + ip)*N1 + IND1 + N1T) *= \
300  pcoefficient.at(ip); \
301  s##QUANT.at((IND2*nplist + ip)*N1 + IND1 + N1T) += \
302  model->k()[nk - nplist + ip] * \
303  s##QUANT.at((IND2*nplist + ip)*N1 + IND1) / \
304  unscaledParameters[model->plist(ip)]; \
305  }
306 
307  s2vecChainRule(x, ix, nxtrue, nx, it, nt);
308  s2vecChainRule(y, iy, nytrue, ny, it, nt);
309  s2vecChainRule(sigmay, iy, nytrue, ny, it, nt);
310  s2vecChainRule(z, iz, nztrue, nz, ie, nmaxevent);
