66  getSteadystateSimulation(rdata, solver, model, it);
67  }

69  } catch (AmiException const &ex2) {
70  /* may be integration failure from AmiSolve, so NewtonFailure

78  throw AmiException("Steady state computation failed to "

80  "number of iterations");
81  throw;
82  }
83  }
84  }
85  run_time = (double)((clock() - starttime) * 1000) / CLOCKS_PER_SEC;
86 
87  /* Compute steady state sensitvities */
88

90  (newton_status == NewtonStatus::newt ||
91  newton_status == NewtonStatus::newt_sim_newt ||
92  model->getSteadyStateSensitivityMode() != SteadyStateSensitivityMode::simulationFSA))

94  newtonSolver->computeNewtonSensis(sx);

137 void SteadystateProblem::applyNewtonsMethod(ReturnData *rdata, Model *model,
138  NewtonSolver *newtonSolver,
139  NewtonStatus steadystate_try) {
140  int i_newtonstep = 0;
141  int ix = 0;
142  double gamma = 1.0;
143  bool compNewStep = TRUE;
144 
145  /* initialize output of linear solver for Newton step */
146  delta.reset();
147 
148  model->fxdot(*t, x, &dx, &xdot);
149 
150  /* Check for relative error, but make sure not to divide by 0!
151  Ensure positivity of the state */
152  x_newton = *x;
153  x_old = *x;
154  xdot_old = xdot;
155 
156  //rdata->newton_numsteps[newton_try - 1] = 0.0;
157  wrms = getWrmsNorm(x_newton, xdot, newtonSolver->atol, newtonSolver->rtol);
158  bool converged = wrms < RCONST(1.0);
159  while (!converged && i_newtonstep < newtonSolver->maxsteps) {
160 
161  /* If Newton steps are necessary, compute the inital search direction */
162  if (compNewStep) {
163  try {
164  delta = xdot;
165  newtonSolver->getStep(steadystate_try == NewtonStatus::newt ? 1
166  : 2,
167  i_newtonstep, &delta);
168  } catch (NewtonFailure const &ex) {
169  rdata->newton_numsteps.at(steadystate_try == NewtonStatus::newt
170  ? 0
171  : 2) = i_newtonstep;
172  throw;
173  } catch (std::exception const &ex) {
174  rdata->newton_numsteps.at(steadystate_try == NewtonStatus::newt
175  ? 0
176  : 2) = i_newtonstep;
177  throw AmiException("Newton solver failed to compute new step: "
178  "%s", ex.what());
179  }
180  }
181 
182  /* Try a full, undamped Newton step */
183  N_VLinearSum(1.0, x_old.getNVector(), gamma, delta.getNVector(), x->getNVector());
184 
185  /* Compute new xdot and residuals */
186  model->fxdot(*t, x, &dx, &xdot);
187  realtype wrms_tmp = getWrmsNorm(x_newton, xdot, newtonSolver->atol,
188  newtonSolver->rtol);
189 
190  if (wrms_tmp < wrms) {
191  /* If new residuals are smaller than old ones, update state */
192  wrms = wrms_tmp;
193  x_old = *x;
194  xdot_old = xdot;
195  /* New linear solve due to new state */
196  compNewStep = TRUE;
197  /* Check residuals vs tolerances */
198  converged = wrms < RCONST(1.0);
199 
200  if (converged) {
201  /* Ensure positivity of the found state */
202  for (ix = 0; ix < model->nx_solver; ix++) {
203  if ((*x)[ix] < 0.0) {
204  (*x)[ix] = 0.0;
205  converged = FALSE;
206  }
207  }
208  } else {
209  /* increase dampening factor (superfluous, if converged) */
210  gamma = fmin(1.0, 2.0 * gamma);
211  }
212  } else {
213  /* Reduce dampening factor */
214  gamma = gamma / 4.0;
215  /* No new linear solve, only try new dampening */
216  compNewStep = FALSE;
217  }
218  /* increase step counter */
219  i_newtonstep++;
220  }
221 
222  /* Set return values */
223  rdata->newton_numsteps.at(steadystate_try == NewtonStatus::newt ? 0 : 2) =
224  i_newtonstep;
225  if (!converged)
226  throw NewtonFailure(AMICI_TOO_MUCH_WORK, "applyNewtonsMethod");
227 }

258 void SteadystateProblem::getSteadystateSimulation(ReturnData *rdata, Solver *solver,
259  Model *model, int it)
260 {
261  /* Loop over steps and check for convergence */
262  bool converged = checkConvergence(solver, model);
263 
264  int steps_newton = 0;
265  while(!converged) {
266  /* One step of ODE integration
267  reason for tout specification:
268  max with 1 ensures correct direction (any positive value would do)
269  multiplication with 10 ensures nonzero difference and should ensure stable computation
270  value is not important for AMICI_ONE_STEP mode, only direction w.r.t. current t
271  */
272  solver->solve(std::max(*t,1.0) * 10, x, &dx, t, AMICI_ONE_STEP);
273 
274  /* Check for convergence */
275  converged = checkConvergence(solver, model);
276  /* increase counter, check for maxsteps */
277  steps_newton++;
278  if (steps_newton >= solver->getMaxSteps() && !converged) {
279  rdata->newton_numsteps.at(static_cast<int>(NewtonStatus::newt_sim) - 1) =
280  steps_newton;
281  throw NewtonFailure(AMICI_TOO_MUCH_WORK, "exceeded maximum number of steps");
282  }
283  }
284  rdata->newton_numsteps.at(static_cast<int>(NewtonStatus::newt_sim) - 1) =
285  steps_newton;
286  if (solver->getSensitivityOrder()>SensitivityOrder::none)
287  solver->getSens(t, sx);
288 }

294
