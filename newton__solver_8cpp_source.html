243 void NewtonSolverSparse::prepareLinearSystem(int ntry, int nnewt) {
253  /* Get sparse Jacobian */
254  model->fJSparse(*t, 0.0, x, &dx, &xdot, Jtmp.slsmat());
255 
256  /* Get factorization of sparse Jacobian */
257  if (!symbolic) /* we only need to perform symbolic factorization once */
258  symbolic = klu_analyze(model->nx_solver, Jtmp.slsmat()->indexptrs,
259  Jtmp.slsmat()->indexvals, &common);
260 
261  if (!symbolic) {
262  throw NewtonFailure(common.status, "klu_analyze");
263  }
264 
265  if (numeric) { /* if numeric we only need to refactor, which can be done
266  very effectively using klu_refactor, see cvode_klu.c for
267  reference for this code*/
268  if (!klu_refactor(Jtmp.slsmat()->indexptrs, Jtmp.slsmat()->indexvals, Jtmp.data(),
269  symbolic, numeric, &common))
270  throw NewtonFailure(common.status, "klu_refactor");
271  /* check cheap estimate of rcond to see if we need to recompute
272  factorization
273  */
274  if (!klu_rcond(symbolic, numeric, &common))
275  throw NewtonFailure(common.status, "klu_rcond");
276 
277  if (common.rcond < SUNRpowerR(UNIT_ROUNDOFF, 2.0 / 3.0)) {
278  /* compute more accurate estimate */
279  if (!klu_condest(Jtmp.slsmat()->indexptrs, Jtmp.data(), symbolic, numeric,
280  &common))
281  throw NewtonFailure(common.status, "klu_condest");
282  if (common.condest > (1.0 / SUNRpowerR(UNIT_ROUNDOFF, 2.0 / 3.0))) {
283  /* if accurate condition estimate exceeds thresholds delete
284  factorization and factor de novo */
285  klu_free_numeric(&numeric, &common);
286  }
287  }
288  }
289 
290  if (!numeric) /* factor de novo if we factor for the first time or deleted
291  previous factorization due to too high condition numbers */
292  numeric = klu_factor(Jtmp.slsmat()->indexptrs, Jtmp.slsmat()->indexvals, Jtmp.data(),
293  symbolic, &common);
294  if (!numeric)
295  throw NewtonFailure(common.status, "klu_factor");
296 } // namespace amici

377 void NewtonSolverIterative::linsolveSPBCG(int ntry,int nnewt, AmiVector *ns_delta) {
389  double rho;
390  double alpha;
391  double omega;
392  double res;
393 
394  xdot = *ns_delta;
395  xdot.minus();
396 
397  // Get the diagonal of the Jacobian for preconditioning
398  model->fJDiag(*t, &ns_Jdiag, 0.0, x, &dx);
399 
400  // Ensure positivity of entries in ns_Jdiag
401  ns_p.set(1.0);
402  N_VAbs(ns_Jdiag.getNVector(), ns_Jdiag.getNVector());
403  N_VCompare(1e-15, ns_Jdiag.getNVector(), ns_tmp.getNVector());
404  N_VLinearSum(-1.0, ns_tmp.getNVector(), 1.0, ns_p.getNVector(), ns_tmp.getNVector());
405  N_VLinearSum(1.0, ns_Jdiag.getNVector(), 1.0, ns_tmp.getNVector(), ns_Jdiag.getNVector());
406 
407  // Initialize for linear solve
408  ns_p.reset();
409  ns_v.reset();
410  ns_delta->reset();
411  ns_tmp.reset();
412  rho = 1.0;
413  omega = 1.0;
414  alpha = 1.0;
415 
416  // can be set to 0 at the moment
417  model->fJv(*t, x, &dx, &xdot, ns_delta, &ns_Jv, 0.0);
418 
419  // ns_r = xdot - ns_Jv;
420  N_VLinearSum(-1.0, ns_Jv.getNVector(), 1.0, xdot.getNVector(), ns_r.getNVector());
421  N_VDiv(ns_r.getNVector(), ns_Jdiag.getNVector(), ns_r.getNVector());
422  res = sqrt(N_VDotProd(ns_r.getNVector(), ns_r.getNVector()));
423  ns_rt = ns_r;
424 
425  for (int i_linstep = 0; i_linstep < maxlinsteps;
426  i_linstep++) {
427  // Compute factors
428  double rho1 = rho;
429  rho = N_VDotProd(ns_rt.getNVector(), ns_r.getNVector());
430  double beta = rho * alpha / (rho1 * omega);
431 
432  // ns_p = ns_r + beta * (ns_p - omega * ns_v);
433  N_VLinearSum(1.0, ns_p.getNVector(), -omega, ns_v.getNVector(), ns_p.getNVector());
434  N_VLinearSum(1.0, ns_r.getNVector(), beta, ns_p.getNVector(), ns_p.getNVector());
435 
436  // ns_v = J * ns_p
437  model->fJv(*t, x, &dx, &xdot, &ns_p, &ns_v, 0.0);
438  N_VDiv(ns_v.getNVector(), ns_Jdiag.getNVector(), ns_v.getNVector());
439 
440  // Compute factor
441  alpha = rho / N_VDotProd(ns_rt.getNVector(), ns_v.getNVector());
442 
443  // ns_h = ns_delta + alpha * ns_p;
444  N_VLinearSum(1.0, ns_delta->getNVector(), alpha, ns_p.getNVector(), ns_h.getNVector());
445  // ns_s = ns_r - alpha * ns_v;
446  N_VLinearSum(1.0, ns_r.getNVector(), -alpha, ns_v.getNVector(), ns_s.getNVector());
447 
448  // ns_t = J * ns_s
449  model->fJv(*t, x, &dx, &xdot, &ns_s, &ns_t, 0.0);
450  N_VDiv(ns_t.getNVector(), ns_Jdiag.getNVector(), ns_t.getNVector());
451 
452  // Compute factor
453  omega = N_VDotProd(ns_t.getNVector(), ns_s.getNVector()) / N_VDotProd(ns_t.getNVector(), ns_t.getNVector());
454 
455  // ns_delta = ns_h + omega * ns_s;
456  N_VLinearSum(1.0, ns_h.getNVector(), omega, ns_s.getNVector(), ns_delta->getNVector());
457  // ns_r = ns_s - omega * ns_t;
458  N_VLinearSum(1.0, ns_s.getNVector(), -omega, ns_t.getNVector(), ns_r.getNVector());
459 
460  // Compute the (unscaled) residual
461  N_VProd(ns_r.getNVector(), ns_Jdiag.getNVector(), ns_r.getNVector());
462  res = sqrt(N_VDotProd(ns_r.getNVector(), ns_r.getNVector()));
463 
464  // Test convergence
465  if (res < atol) {
466  // Write number of steps needed
467  rdata->newton_numlinsteps[(ntry - 1) * maxsteps +
468  nnewt] = i_linstep + 1;
469 
470  // Return success
471  return;
472  }
473 
474  // Scale back
475  N_VDiv(ns_r.getNVector(), ns_Jdiag.getNVector(), ns_r.getNVector());
476  }
477  throw NewtonFailure(AMICI_CONV_FAILURE, "linsolveSPBCG");
478 }
