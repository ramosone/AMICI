75 void ForwardProblem::workForwardProblem() {
76 
77  try {
78  solver->setup(&x, &dx, &sx, &sdx, model);
79  } catch (std::exception const& ex) {
80  throw AmiException("AMICI setup failed:\n(%s)",ex.what());
81  } catch (...) {
82  throw AmiException("AMICI setup failed due to an unknown error");
83  }
84 
85  model->fx_rdata(&x_rdata, &x);
86  if(solver->getSensitivityOrder() >= SensitivityOrder::first) {
87  model->fsx_rdata(&sx_rdata, &sx);
88  }
89 
90  if(edata){
91  rdata->initializeObjectiveFunction();
92  }
93 
94  /* if preequilibration is necessary, start Newton solver */
95  if (solver->getNewtonPreequilibration() || (edata && !edata->fixedParametersPreequilibration.empty())) {
96  handlePreequilibration();
97  } else {
98  model->fx_rdata(&x_rdata, &x);
99  rdata->x0 = std::move(x_rdata.getVector());
100  if (solver->getSensitivityMethod() == SensitivityMethod::forward &&
101  solver->getSensitivityOrder() >= SensitivityOrder::first) {
102  model->fsx_rdata(&sx_rdata, &sx);
103  for (int ix = 0; ix < rdata->nx; ix++) {
104  for (int ip = 0; ip < model->nplist(); ip++)
105  rdata->sx0[ip*rdata->nx + ix] = sx_rdata.at(ix,ip);
106  }
107  }
108  }
109 
110  int ncheck = 0; /* the number of (internal) checkpoints stored so far */
111 
112  /* perform presimulation if necessary */
113  if (edata && edata->t_presim > 0)
114  handlePresimulation(&ncheck);
115 
116  /* loop over timepoints */
117  for (int it = 0; it < model->nt(); it++) {
118  auto nextTimepoint = model->t(it);
119 
120  if (solver->getSensitivityMethod() == SensitivityMethod::forward &&
121  solver->getSensitivityOrder() >= SensitivityOrder::first) {
122  solver->setStopTime(nextTimepoint);
123  }
124 
125  if (nextTimepoint > model->t0()) {
126  while (t < nextTimepoint) {
127  if (model->nx_solver == 0) {
128  t = nextTimepoint;
129  continue;
130  }
131 
132  if (std::isinf(nextTimepoint)) {
133  SteadystateProblem sstate = SteadystateProblem(&t, &x, &sx);
134  sstate.workSteadyStateProblem(rdata, solver, model, it);
135  } else {
136  int status;
137  if (solver->getSensitivityMethod() == SensitivityMethod::adjoint &&
138  solver->getSensitivityOrder() >= SensitivityOrder::first) {
139  status = solver->solveF(RCONST(nextTimepoint), &x, &dx,
140  &(t), AMICI_NORMAL, &ncheck);
141  } else {
142  status = solver->solve(RCONST(nextTimepoint), &x, &dx,
143  &(t), AMICI_NORMAL);
144  }
145 
146  if (status == AMICI_ILL_INPUT) {
147  /* clustering of roots => turn off rootfinding */
148  solver->turnOffRootFinding();
149  } else if (status == AMICI_ROOT_RETURN) {
150  handleEvent(&tlastroot, false);
151  }
152  }
153  }
154  }
155  handleDataPoint(it);
156  }
157 
158  /* fill events */
159  if (model->nz > 0) {
160  getEventOutput();
161  }
162 
163  // set likelihood
164  if (!edata) {
165  rdata->invalidateLLH();
166  rdata->invalidateSLLH();
167  }
168 
169  storeJacobianAndDerivativeInReturnData();
170 }

172 void ForwardProblem::handlePreequilibration()
173 {
174  // Are there dedicated condition preequilibration parameters provided?
175  bool overrideFixedParameters = edata && !edata->fixedParametersPreequilibration.empty();
176 
177  std::vector<realtype> originalFixedParameters; // to restore after pre-equilibration
178 
179  if(overrideFixedParameters) {
180  if(edata->fixedParametersPreequilibration.size() != (unsigned) model->nk())
181  throw AmiException("Number of fixed parameters (%d) in model does not match preequilibration parameters in ExpData (%zd).",
182  model->nk(), edata->fixedParametersPreequilibration.size());
183  originalFixedParameters = model->getFixedParameters();
184  model->setFixedParameters(edata->fixedParametersPreequilibration);
185  }
186 
187  // pre-equilibrate
188  SteadystateProblem sstate = SteadystateProblem(&t, &x, &sx);
189 
190  sstate.workSteadyStateProblem(rdata, solver, model, -1);
191 
192  if(overrideFixedParameters) { // Restore
193  model->setFixedParameters(originalFixedParameters);
194  }
195 
196  updateAndReinitStatesAndSensitivities();
197 }

225 void ForwardProblem::handlePresimulation(int *ncheck)
226 {
227  // Are there dedicated condition preequilibration parameters provided?
228  bool overrideFixedParameters = edata && !edata->fixedParametersPresimulation.empty();
229 
230  std::vector<realtype> originalFixedParameters; // to restore after pre-equilibration
231 
232  if(overrideFixedParameters) {
233  if(edata->fixedParametersPresimulation.size() != (unsigned) model->nk())
234  throw AmiException("Number of fixed parameters (%d) in model does not match presimulation parameters in ExpData (%zd).",
235  model->nk(), edata->fixedParametersPresimulation.size());
236  originalFixedParameters = model->getFixedParameters();
237  model->setFixedParameters(edata->fixedParametersPresimulation);
238  }
239  t = model->t0() - edata->t_presim;
240  updateAndReinitStatesAndSensitivities();
241 
242  if (solver->getSensitivityMethod() == SensitivityMethod::adjoint &&
243  solver->getSensitivityOrder() >= SensitivityOrder::first) {
244  solver->solveF(RCONST(model->t0()), &x, &dx,
245  &(t), AMICI_NORMAL, ncheck);
246  } else {
247  solver->solve(RCONST(model->t0()), &x, &dx,
248  &(t), AMICI_NORMAL);
249  }
250 
251  if(overrideFixedParameters) {
252  model->setFixedParameters(originalFixedParameters);
253  }
254  t = model->t0();
255  updateAndReinitStatesAndSensitivities();
256 }

262 void ForwardProblem::handleEvent(realtype *tlastroot, const bool seflag) {
269  /* store heaviside information at event occurence */
270  model->froot(t, &x, &dx, rootvals.data());
271 
272  if (!seflag) {
273  solver->getRootInfo(rootsfound.data());
274  }
275 
276  if (iroot < model->nMaxEvent() * model->ne) {
277  std::copy(rootsfound.begin(), rootsfound.end(), &rootidx[iroot * model->ne]);
278  }
279 
280  rvaltmp = rootvals;
281 
282  if (!seflag) {
283  /* only extract in the first event fired */
284  if (solver->getSensitivityOrder() >= SensitivityOrder::first &&
285  solver->getSensitivityMethod() == SensitivityMethod::forward) {
286  solver->getSens(&(t), &sx);
287  }
288 
289  /* only check this in the first event fired, otherwise this will always
290  * be true */
291  if (t == *tlastroot) {
292  throw AmiException("AMICI is stuck in an event, as the initial"
293  "step-size after the event is too small. To fix "
294  "this, increase absolute and relative tolerances!");
295  }
296  *tlastroot = t;
297  }
298 
299  if(model->nz>0)
300  getEventOutput();
301 
302  /* if we need to do forward sensitivities later on we need to store the old
303  * x and the old xdot */
304  if (solver->getSensitivityOrder() >= SensitivityOrder::first) {
305  /* store x and xdot to compute jump in sensitivities */
306  x_old = x;
307  if (solver->getSensitivityMethod() == SensitivityMethod::forward) {
308  model->fxdot(t, &x, &dx, &xdot);
309  xdot_old = xdot;
310  dx_old = dx;
311 
312  /* compute event-time derivative only for primary events, we get
313  * into trouble with multiple simultaneously firing events here (but
314  * is this really well defined then?), in that case just use the
315  * last ie and hope for the best. */
316  if (!seflag) {
317  for (int ie = 0; ie < model->ne; ie++) {
318  if (rootsfound.at(ie) == 1) {
319  /* only consider transitions false -> true */
320  model->fstau(t, ie, &x, &sx);
321  }
322  }
323  }
324  } else if (solver->getSensitivityMethod() == SensitivityMethod::adjoint) {
325  /* store x to compute jump in discontinuity */
326  if (iroot < model->nMaxEvent() * model->ne) {
327  x_disc[iroot] = x;
328  xdot_disc[iroot] = xdot;
329  xdot_old_disc[iroot] = xdot_old;
330  }
331  }
332  }
333 
334  model->updateHeaviside(rootsfound);
335 
336  applyEventBolus();
337 
338  if (iroot < model->nMaxEvent() * model->ne) {
339  discs[iroot] = t;
340  ++iroot;
341  } else {
342  warnMsgIdAndTxt("AMICI:mex:TOO_MUCH_EVENT",
343  "Event was recorded but not reported as the number of "
344  "occured events exceeded (nmaxevents)*(number of "
345  "events in model definition)!");
346  solver->reInit(t, &x, &dx); /* reinitialise so that we can continue in peace */
347  return;
348  }
349 
350  if (solver->getSensitivityOrder() >= SensitivityOrder::first
351  && solver->getSensitivityMethod() == SensitivityMethod::forward) {
352  /* compute the new xdot */
353  model->fxdot(t, &x, &dx, &xdot);
354  applyEventSensiBolusFSA();
355  }
356 
357  int secondevent = 0;
358 
359  /* check whether we need to fire a secondary event */
360  model->froot(t, &x, &dx, rootvals.data());
361  for (int ie = 0; ie < model->ne; ie++) {
362  /* the same event should not trigger itself */
363  if (rootsfound.at(ie) == 0) {
364  /* check whether there was a zero-crossing */
365  if (0 > rvaltmp.at(ie) * rootvals.at(ie)) {
366  if (rvaltmp.at(ie) < rootvals.at(ie)) {
367  rootsfound.at(ie) = 1;
368  } else {
369  rootsfound.at(ie) = -1;
370  }
371  secondevent++;
372  } else {
373  rootsfound.at(ie) = 0;
374  }
375  } else {
376  /* don't fire the same event again */
377  rootsfound.at(ie) = 0;
378  }
379  }
380  /* fire the secondary event */
381  if (secondevent > 0) {
382  handleEvent(tlastroot, TRUE);
383  }
384 
385  /* only reinitialise in the first event fired */
386  if (!seflag) {
387  solver->reInit(t, &x, &dx);
388 
389  /* make time derivative consistent */
390  solver->calcIC(t, &x, &dx);
391 
392  if (solver->getSensitivityOrder() >= SensitivityOrder::first) {
393  if (solver->getSensitivityMethod() == SensitivityMethod::forward) {
394  solver->sensReInit(&sx, &sdx);
395  }
396  }
397  }

413 void ForwardProblem::getEventOutput() {
418  if (t == model->gett(model->nt() - 1,rdata)) {
419  // call from fillEvent at last timepoint
420  model->froot(t, &x, &dx, rootvals.data());
421  }
422 
423  /* EVENT OUTPUT */
424  for (int ie = 0; ie < model->ne; ie++) {
425  /* only look for roots of the rootfunction not discontinuities */
426  if (nroots.at(ie) >= model->nMaxEvent())
427  continue;
428 
429  /* only consider transitions false -> true or event filling */
430  if (rootsfound.at(ie) != 1 && t != model->gett(model->nt() - 1, rdata)) {
431  continue;
432  }
433 
434  model->fz(nroots.at(ie), ie, t, &x, rdata);
435 
436  if (edata) {
437  model->fsigmaz(t, ie, nroots.data(), rdata, edata);
438  model->fJz(nroots.at(ie), rdata, edata);
439 
440  if (t == model->gett(model->nt() - 1,rdata)) {
441  // call from fillEvent at last
442  // timepoint, add regularization
443  // based on rz
444  model->frz(nroots.at(ie), ie, t, &x, rdata);
445  model->fJrz(nroots.at(ie), rdata, edata);
446  }
447  }
448 
449  if (solver->getSensitivityOrder() >= SensitivityOrder::first) {
450  prepEventSensis(ie);
451  if (solver->getSensitivityMethod() == SensitivityMethod::forward) {
452  getEventSensisFSA(ie);
453  }
454  }
455 
456  nroots.at(ie)++;
457  }
458 
459  if (t == model->gett(model->nt() - 1, rdata)) {
460  // call from fillEvent at last timepoint
461  // loop until all events are filled
462  if(std::any_of(nroots.cbegin(), nroots.cend(), [&](int curNRoots){ return curNRoots < model->nMaxEvent(); }))
463  getEventOutput();
464  }
