45 void Model::fsJy(const int it, const std::vector<realtype>& dJydx, const AmiVectorArray *sx, ReturnData *rdata) {
46 
47  // Compute dJydx*sx for current 'it'
48  // dJydx rdata->nt x nJ x nx_solver
49  // sx rdata->nt x nx_solver x nplist()
50  std::vector<realtype> multResult(nJ * nplist(), 0);
51  sx->flatten_to_vector(this->sx);
52 
53  // C := alpha*op(A)*op(B) + beta*C,
54  amici_dgemm(BLASLayout::colMajor, BLASTranspose::noTrans, BLASTranspose::noTrans, nJ,
55  nplist(), nx_solver, 1.0, &dJydx.at(it*nJ*nx_solver), nJ, this->sx.data(), nx_solver, 0.0,
56  multResult.data(), nJ);
57 
58  // multResult nJ x nplist()
59  // dJydp nJ x nplist()
60  // dJydxTmp nJ x nx_solver
61  // sx nx_solver x nplist()
62 
63  // sJy += multResult + dJydp
64  for (int iJ = 0; iJ < nJ; ++iJ) {
65  if (iJ == 0)
66  for (int ip = 0; ip < nplist(); ++ip)
67  rdata->sllh.at(ip) -= multResult.at(ip * nJ) + dJydp.at(ip * nJ);
68  else
69  for (int ip = 0; ip < nplist(); ++ip)
70  rdata->s2llh.at((iJ - 1) + ip * (nJ - 1)) -=
71  multResult.at(iJ + ip * nJ) + dJydp.at(iJ + ip * nJ);

75 void Model::fdJydp(const int it, ReturnData *rdata, const ExpData *edata) {
76 
77  // dJydy nJ x nytrue x ny
78  // dydp nplist * ny
79  // dJydp nplist x nJ
80  // dJydsigma
81 
82  getmy(it,edata);
83  std::fill(dJydp.begin(), dJydp.end(), 0.0);
84 
85  for (int iyt = 0; iyt < nytrue; ++iyt) {
86  if (isNaN(my.at(iyt)))
87  continue;
88 
89  // dJydp = 1.0 * dJydp + 1.0 * dJydy * dydp
90  amici_dgemm(BLASLayout::colMajor, BLASTranspose::noTrans, BLASTranspose::noTrans,
91  nJ, nplist(), ny,
92  1.0, &dJydy.at(iyt*nJ*ny), nJ,
93  dydp.data(), ny,
94  1.0, dJydp.data(), nJ);
95 
96  // dJydp = 1.0 * dJydp + 1.0 * dJydsigma * dsigmaydp
97  amici_dgemm(BLASLayout::colMajor, BLASTranspose::noTrans, BLASTranspose::noTrans,
98  nJ, nplist(), ny,
99  1.0, &dJydsigma.at(iyt*nJ*ny), nJ,
100  dsigmaydp.data(), ny,
101  1.0, dJydp.data(), nJ);
102  }
103 
104  if (rdata->sensi_meth != SensitivityMethod::adjoint)
105  return;
106 
107  if(!ny)
108  return;
109 
110  for (int iJ = 0; iJ < nJ; iJ++) {
111  for (int ip = 0; ip < nplist(); ip++) {
112  if (iJ == 0) {
113  rdata->sllh.at(ip) -= dJydp[ip * nJ];
114  } else {
115  rdata->s2llh.at((iJ - 1) + ip * (nJ - 1)) -=
116  dJydp[iJ + ip * nJ];
117  }
118  }
119  }
