69 def getHdf5Config():
70  if pkgconfig:
71  h5pkgcfg = pkgconfig.parse('hdf5')
72  # NOTE: Cannot use pkgconfig.exists('hdf5f'), since this is true
73  # althoughno libraries or include dirs are available
74  h5pkgcfg['found'] = 'include_dirs' in h5pkgcfg \
75  and h5pkgcfg['include_dirs']
76  if h5pkgcfg['found']:
77  return h5pkgcfg
78 
79  h5pkgcfg = {'include_dirs': [],
80  'library_dirs': [],
81  'libraries': [],
82  'define_macros': []
83  }
84 
85  # try for hdf5 in standard locations
86  hdf5_include_dir_hints = [
87  '/usr/include/hdf5/serial',
88  '/usr/local/include',
89  '/usr/include', # travis ubuntu xenial
90  '/usr/local/Cellar/hdf5/1.10.2_1/include' # travis macOS
91  ]
92  hdf5_library_dir_hints = [
93  '/usr/lib/x86_64-linux-gnu/', # travis ubuntu xenial
94  '/usr/lib/x86_64-linux-gnu/hdf5/serial',
95  '/usr/local/lib',
96  '/usr/local/Cellar/hdf5/1.10.2_1/lib' # travis macOS
97  ]
98 
99  for hdf5_include_dir_hint in hdf5_include_dir_hints:
100  hdf5_include_dir_found = os.path.isfile(
101  os.path.join(hdf5_include_dir_hint, 'hdf5.h'))
102  if hdf5_include_dir_found:
103  print('hdf5.h found in %s' % hdf5_include_dir_hint)
104  h5pkgcfg['include_dirs'] = [hdf5_include_dir_hint]
105  break
106 
107  for hdf5_library_dir_hint in hdf5_library_dir_hints:
108  hdf5_library_dir_found = os.path.isfile(
109  os.path.join(hdf5_library_dir_hint, 'libhdf5.a'))
110  if hdf5_library_dir_found:
111  print('libhdf5.a found in %s' % hdf5_library_dir_hint)
112  h5pkgcfg['library_dirs'] = [hdf5_library_dir_hint]
113  break
114  h5pkgcfg['found'] = hdf5_include_dir_found and hdf5_library_dir_found
115 
116  return h5pkgcfg
