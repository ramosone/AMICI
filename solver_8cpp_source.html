50 void Solver::setup(AmiVector *x, AmiVector *dx, AmiVectorArray *sx, AmiVectorArray *sdx, Model *model) {
51  bool computeSensitivities = sensi >= SensitivityOrder::first
52  && model->nx_solver > 0;
53  model->initialize(x, dx, sx, sdx, computeSensitivities);
54 
55  /* Create solver memory object */
56  allocateSolver();
57  if (!solverMemory)
58  throw AmiException("Failed to allocated solver memory!");
59 
60  /* Initialize CVodes/IDAs solver*/
61  init(x, dx, model->t0());
62 
63  applyTolerances();
64 
65  /* Set optional inputs */
66  setErrHandlerFn();
67  /* attaches userdata*/
68  setUserData(model);
69  /* specify maximal number of steps */
70  setMaxNumSteps(maxsteps);
71  /* activates stability limit detection */
72  setStabLimDet(stldet);
73 
74  rootInit(model->ne);
75 
76  initializeLinearSolver(model);
77 
78  if (computeSensitivities) {
79  auto plist = model->getParameterList();
80 
81  if (sensi_meth == SensitivityMethod::forward && !plist.empty()) {
82  /* Set sensitivity analysis optional inputs */
83  auto par = model->getUnscaledParameters();
84 
85  /* Activate sensitivity calculations */
86  sensInit1(sx, sdx, plist.size());
87  setSensParams(par.data(), nullptr, plist.data());
88 
89  applyTolerancesFSA();
90  } else if (sensi_meth == SensitivityMethod::adjoint) {
91  /* Allocate space for the adjoint computation */
92  adjInit();
93  }
94  }
95 
96  setId(model);
97  setSuppressAlg(true);
98  /* calculate consistent DAE initial conditions (no effect for ODE) */
99  if(model->nt()>1)
100  calcIC(model->t(1), x, dx);
101 }

103 void Solver::setupB(BackwardProblem *bwd, Model *model) {
104  if (!solverMemory)
105  throw AmiException("Solver for the forward problem must be setup first");
106 
107  /* write initial conditions */
108  std::vector<realtype> dJydx = bwd->getdJydx();
109  AmiVector *xB = bwd->getxBptr();
110  xB->reset();
111  for (int ix = 0; ix < model->nxtrue_solver; ++ix)
112  for (int iJ = 0; iJ < model->nJ; ++iJ)
113  xB->at(ix + iJ * model->nxtrue_solver) +=
114  dJydx.at(iJ + ( ix + (model->nt() - 1) * model->nx_solver ) * model->nJ);
115  bwd->getdxBptr()->reset();
116  bwd->getxQBptr()->reset();
117 
118  /* allocate memory for the backward problem */
119  allocateSolverB(bwd->getwhichptr());
120 
121  /* initialise states */
122  binit(bwd->getwhich(), bwd->getxBptr(), bwd->getdxBptr(), bwd->gett());
123 
124  /* Attach user data */
125  setUserDataB(bwd->getwhich(), model);
126 
127  /* Number of maximal internal steps */
128  setMaxNumStepsB(bwd->getwhich(), (maxstepsB == 0) ? maxsteps * 100 : maxstepsB);
129 
130  initializeLinearSolverB(model, bwd->getwhich());
131 
132  /* Initialise quadrature calculation */
133  qbinit(bwd->getwhich(), bwd->getxQBptr());
134 
135  applyTolerancesASA(bwd->getwhich());
136  applyQuadTolerancesASA(bwd->getwhich());
137 
138  setStabLimDetB(bwd->getwhich(), stldet);
139 }

141 void Solver::wrapErrHandlerFn(int error_code, const char *module,
142  const char *function, char *msg, void * /*eh_data*/) {
143  char buffer[250];
144  char buffid[250];
145  sprintf(buffer, "AMICI ERROR: in module %s in function %s : %s ", module,
146  function, msg);
147  switch (error_code) {
148  case 99:
149  sprintf(buffid, "AMICI:mex:%s:%s:WARNING", module, function);
150  break;
151 
152  case -1:
153  sprintf(buffid, "AMICI:mex:%s:%s:TOO_MUCH_WORK", module, function);
154  break;
155 
156  case -2:
157  sprintf(buffid, "AMICI:mex:%s:%s:TOO_MUCH_ACC", module, function);
158  break;
159 
160  case -3:
161  sprintf(buffid, "AMICI:mex:%s:%s:ERR_FAILURE", module, function);
162  break;
163 
164  case -4:
165  sprintf(buffid, "AMICI:mex:%s:%s:CONV_FAILURE", module, function);
166  break;
167 
168  default:
169  sprintf(buffid, "AMICI:mex:%s:%s:OTHER", module, function);
170  break;
171  }
172 
173  warnMsgIdAndTxt(buffid, buffer);
174 }
