68  void storeBacktrace(const int nMaxFrames) {
69  std::ostringstream trace_buf;
70 
71  #ifdef PLATFORM_WINDOWS
72 
73  trace_buf << "stacktrace not available on windows platforms\n";
74 
75  #else
76 
77  void *callstack[nMaxFrames];
78  char buf[1024];
79  int nFrames = backtrace(callstack, nMaxFrames);
80  char **symbols = backtrace_symbols(callstack, nFrames);
81 
82  for (int i = 2; i < nFrames;
83  i++) { // start at 2 to omit AmiException and storeBacktrace
84  // call
85  Dl_info info;
86  if (dladdr(callstack[i], &info) && info.dli_sname) {
87  char *demangled = NULL;
88  int status = -1;
89  if (info.dli_sname[0] == '_')
90  demangled = abi::__cxa_demangle(info.dli_sname, NULL, 0,
91  &status);
92  snprintf(buf, sizeof(buf), "%-3d %*p %s + %zd\n", i - 2,
93  int(2 + sizeof(void *) * 2), callstack[i],
94  status == 0 ? demangled
95  : info.dli_sname == 0 ? symbols[i]
96  : info.dli_sname,
97  static_cast<ssize_t>((char *)callstack[i] -
98  (char *)info.dli_saddr));
99  free(demangled);
100  } else {
101  snprintf(buf, sizeof(buf), "%-3d %*p %s\n", i - 2,
102  int(2 + sizeof(void *) * 2), callstack[i],
103  symbols[i]);
104  }
105  trace_buf << buf;
106  }
107  free(symbols);
108  if (nFrames == nMaxFrames)
109  trace_buf << "[truncated]\n";
110 
111  #endif
112 
113  snprintf(trace, sizeof(trace), "%s", trace_buf.str().c_str());
114  }
