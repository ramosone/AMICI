60 ExpData::ExpData(ReturnData const& rdata, std::vector<realtype> sigma_y,
61  std::vector<realtype> sigma_z)
62  : ExpData(rdata.nytrue, rdata.nztrue, rdata.nmaxevent, rdata.ts)
63 {
64  if (sigma_y.size() != (unsigned) nytrue_ && sigma_y.size() != (unsigned) nytrue_*nt())
65  throw AmiException("Dimension of sigma_y must be %d or %d, was %d", nytrue_, nytrue_*nt(), sigma_y.size());
66 
67  if (sigma_z.size() != (unsigned) nztrue_ && sigma_z.size() != (unsigned) nztrue_*nmaxevent_)
68  throw AmiException("Dimension of sigma_z must be %d or %d, was %d", nztrue_, nztrue_*nmaxevent_, sigma_z.size());
69 
70  std::random_device rd{};
71  std::mt19937 gen{rd()};
72 
73  realtype sigma;
74 
75  checkSigmaPositivity(sigma_y, "sigma_y");
76  checkSigmaPositivity(sigma_z, "sigma_z");
77 
78  for (int iy = 0; iy < nytrue_; ++iy) {
79  for (int it = 0; it < nt(); ++it) {
80  sigma = sigma_y.size() == (unsigned) nytrue_ ? sigma_y.at(iy) : sigma_y.at(iy + nytrue_ * it);
81  std::normal_distribution<> e{0, sigma};
82  observedData.at(iy + nytrue_ * it) = rdata.y.at(iy + rdata.ny * it) + e(gen);
83  observedDataStdDev.at(iy + nytrue_ * it) = sigma;
84  }
85  }
86 
87  for (int iz = 0; iz < nztrue_; ++iz) {
88  for (int ie = 0; ie < nmaxevent_; ++ie) {
89  sigma = sigma_z.size() == (unsigned) nztrue_ ? sigma_z.at(iz) : sigma_z.at(iz + nztrue_ * ie);
90  std::normal_distribution<> e{0, sigma};
91  observedEvents.at(iz + rdata.nztrue * ie) = rdata.z.at(iz + rdata.nz * ie) + e(gen);
92  observedDataStdDev.at(iz + rdata.nztrue * ie) = sigma;
93  }
94  }

338 void ConditionContext::applyCondition(const ExpData *edata)
339 {
340  if(!edata)
341  return;
342 
343  if(!edata->fixedParameters.empty()) {
344  // fixed parameter in model are superseded by those provided in edata
345  if(edata->fixedParameters.size() != (unsigned) model->nk())
346  throw AmiException("Number of fixed parameters (%d) in model does not match ExpData (%zd).",
347  model->nk(), edata->fixedParameters.size());
348  model->setFixedParameters(edata->fixedParameters);
349  }
350 
351  if(edata->nt()) {
352  // fixed parameter in model are superseded by those provided in edata
353  model->setTimepoints(edata->getTimepoints());
354  }
355 }
