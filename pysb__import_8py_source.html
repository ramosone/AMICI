580 def _compute_target_index(cl_prototypes, ODE):
581  possible_indices = list(set(list(itertools.chain(*[
582  cl_prototypes[monomer]['possible_indices']
583  for monomer in cl_prototypes
584  ]))))
585 
586  # Note: currently this function is supposed to also count appearances in
587  # expressions. However, expressions are currently still empty as they
588  # are also populated from conservation laws. In case there are many
589  # state heavy expressions in the model (should not be the case for mass
590  # action kinetics). This may lead to suboptimal results and could improved.
591  # As this would require substantial code shuffling, this will only be
592  # fixed if this becomes an actual problem
593  appearance_counts = ODE.get_appearance_counts(possible_indices)
594 
595  # in this initial guess we ignore the cost of having cyclic dependencies
596  # between conservation laws
597  for monomer in cl_prototypes:
598  prototype = cl_prototypes[monomer]
599  # extract monomer specific appearance counts
600  prototype['appearance_counts'] = \
601  [
602  appearance_counts[possible_indices.index(idx)]
603  for idx in prototype['possible_indices']
604  ]
605  # select target index as possible index with minimal appearance count
606  if len(prototype['appearance_counts']) == 0:
607  raise Exception(f'Failed to compute conservation law for monomer '
608  f'{monomer}')
609 
610  idx = np.argmin(prototype['appearance_counts'])
611 
612  # remove entries from possible indices and appearance counts so we
613  # do not consider them again in later iterations
614  prototype['target_index'] = prototype['possible_indices'].pop(idx)
615  prototype['appearance_count'] = prototype['appearance_counts'].pop(idx)
616 
617  # this is only an approximation as the effective species count
618  # of other conservation laws may also be affected by the chosen
619  # target index. As long as the number of unique monomers in
620  # multimers has a low upper bound and the species count does not
621  # vary too much across conservation laws, this approximation
622  # should be fine
623  prototype['fillin'] = \
624  prototype['appearance_count'] * prototype['species_count']
625 
626  # we might end up with the same index for multiple monomers, so loop until
627  # we have a set of unique target indices
628  while not _cl_prototypes_are_valid(cl_prototypes):
629  _greedy_target_index_update(cl_prototypes)

754 def _greedy_target_index_update(cl_prototypes):
755 
756  target_indices = get_target_indices(cl_prototypes)
757 
758  for monomer, prototype in cl_prototypes.items():
759  if target_indices.count(prototype['target_index']) > 1 or \
760  _cl_has_cycle(monomer, cl_prototypes):
761  # compute how much fillin the next best target_index would yield
762 
763  # we exclude already existing target indices to avoid that
764  # updating the target index removes uniqueness from already unique
765  # target indices, this may slightly reduce chances of finding a
766  # solution but prevents infinite loops
767  for target_index in list(set(target_indices)):
768  try:
769  local_idx = prototype['possible_indices'].index(
770  target_index
771  )
772  except ValueError:
773  local_idx = None
774 
775  if local_idx:
776  del prototype['possible_indices'][local_idx]
777  del prototype['appearance_counts'][local_idx]
778 
779  if len(prototype['possible_indices']) == 0:
780  prototype['diff_fillin'] = -1
781  continue
782 
783  idx = np.argmin(prototype['appearance_counts'])
784 
785  prototype['local_index'] = idx
786  prototype['alternate_target_index'] = \
787  prototype['possible_indices'][idx]
788  prototype['alternate_appearance_count'] = \
789  prototype['appearance_counts'][idx]
790 
791  prototype['alternate_fillin'] = \
792  prototype['alternate_appearance_count'] \
793  * prototype['species_count']
794 
795  prototype['diff_fillin'] = \
796  prototype['alternate_fillin'] - prototype['fillin']
797  else:
798  prototype['diff_fillin'] = -1
799 
800  if all(
801  prototype['diff_fillin'] == -1
802  for prototype in cl_prototypes.values()
803  ):
804  raise Exception('Could not compute a valid set of conservation '
805  'laws for this model!')
806 
807  # this puts prototypes with high diff_fillin last
808  cl_prototypes = sorted(
809  cl_prototypes.items(), key=lambda kv: kv[1]['diff_fillin']
810  )
811  cl_prototypes = {
812  proto[0]: proto[1]
813  for proto in cl_prototypes
814  }
815 
816  for monomer in cl_prototypes:
817  prototype = cl_prototypes[monomer]
818  # we check that we
819  # A) have an alternative index computed, i.e. that
820  # that monomer originally had a non-unique target_index
821  # B) that the target_index still is not unique or part of a cyclic
822  # dependency. due to the sorting, this will always be the monomer
823  # with the highest diff_fillin (note that the target index counts
824  # are recomputed on the fly)
825 
826 
827  if prototype['diff_fillin'] > -1 \
828  and (
829  get_target_indices(cl_prototypes).count(
830  prototype['target_index']
831  ) > 1
832  or _cl_has_cycle(monomer, cl_prototypes)
833  ):
834  prototype['fillin'] = prototype['alternate_fillin']
835  prototype['target_index'] = prototype['alternate_target_index']
836  prototype['appearance_count'] = \
837  prototype['alternate_appearance_count']
838 
839  del prototype['possible_indices'][prototype['local_index']]
840  del prototype['appearance_counts'][prototype['local_index']]

876 def _construct_conservation_from_prototypes(cl_prototypes, model):
877  conservation_laws = []
878  for monomer_name in cl_prototypes:
879  target_index = cl_prototypes[monomer_name]['target_index']
880 
881  # T = sum_i(a_i * x_i)
882  # x_j = (T - sum_i≠j(a_i * x_i))/a_j
883  # law: sum_i≠j(a_i * x_i))/a_j
884  # state: x_j
885  target_expression = sum(
886  sp.Symbol(f'__s{ix}')
887  * extract_monomers(specie).count(monomer_name)
888  for ix, specie in enumerate(model.species)
889  if ix != target_index
890  ) / extract_monomers(model.species[target_index]).count(monomer_name)
891  # normalize by the stoichiometry of the target species
892  target_state = sp.Symbol(f'__s{target_index}')
893  # = x_j
894 
895  total_abundance = sp.Symbol(f'tcl__s{target_index}')
896  # = T/a_j
897 
898  state_expr = total_abundance - target_expression
899  # x_j = T/a_j - sum_i≠j(a_i * x_i)/a_j
900 
901  abundance_expr = target_expression + target_state
902  # T/a_j = sum_i≠j(a_i * x_i)/a_j + x_j
903 
904  conservation_laws.append({
905  'state': target_state,
906  'total_abundance': total_abundance,
907  'state_expr': state_expr,
908  'abundance_expr': abundance_expr,
909  })
910 
911  return conservation_laws
