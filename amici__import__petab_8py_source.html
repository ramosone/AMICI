91 def get_fixed_parameters(condition_file_name, sbml_model,
92  constant_species_to_parameters=True):
93 
94  fixed_parameters_df = petab.get_condition_df(condition_file_name)
95  print(f'Condition table: {fixed_parameters_df.shape}')
96 
97  # column names are model parameter names that should be made constant
98  fixed_parameters = list(fixed_parameters_df.columns)
99  try:
100  fixed_parameters.remove('conditionName')
101  except ValueError:
102  pass
103  # must be unique
104  assert(len(fixed_parameters) == len(set(fixed_parameters)))
105 
106  if constant_species_to_parameters:
107  # Turn species which are marked constant in the SBML model into
108  # parameters
109  constant_species = \
110  petab.constant_species_to_parameters(sbml_model)
111 
112  print("Constant species converted to parameters",
113  len(constant_species))
114  print("Non-constant species", len(sbml_model.getListOfSpecies()))
115 
116  # ... and append them to the list of fixed_parameters
117  for species in constant_species:
118  if species not in fixed_parameters:
119  fixed_parameters.append(species)
120 
121  # Ensure mentioned parameters exist in the model. Remove additional ones
122  # from list
123  for fixed_parameter in fixed_parameters[:]:
124  # check global parameters
125  if not sbml_model.getParameter(fixed_parameter) \
126  and not sbml_model.getSpecies(fixed_parameter):
127  print(f"{Fore.YELLOW}Parameter or species '{fixed_parameter}' "
128  "provided in condition table but not present in model.")
129  fixed_parameters.remove(fixed_parameter)
130 
131  return fixed_parameters

135 def import_model(sbml_file: str,
136  condition_file: str,
137  measurement_file: str = None,
138  model_name: str = None,
139  model_output_dir: str = None,
140  verbose: bool = True,
141  allow_reinit_fixpar_initcond: bool = False,
142  **kwargs):
143 
144  if model_name is None:
145  model_name = os.path.splitext(os.path.split(sbml_file)[-1])[0]
146 
147  if model_output_dir is None:
148  model_output_dir = os.path.join(os.getcwd(), model_name)
149 
150  if verbose:
151  print(f"{Fore.GREEN}Importing model '{sbml_file}' using fixed "
152  f"parameters file '{condition_file}'")
153  print(f"{Fore.GREEN}Model name is '{model_name}' Writing model code "
154  f"to '{model_output_dir}'")
155 
156  sbml_importer = amici.SbmlImporter(sbml_file)
157  sbml_model = sbml_importer.sbml
158 
159  show_model_info(sbml_model)
160 
161  observables = petab.get_observables(sbml_importer.sbml, remove=True)
162 
163  sigmas = petab.get_sigmas(sbml_importer.sbml, remove=True)
164 
165  measurement_df = petab.get_measurement_df(measurement_file)
166  if 'observableTransformation' in measurement_df \
167  and not np.issubdtype(
168  measurement_df.observableTransformation.dtype, np.number) \
169  and np.any(measurement_df.observableTransformation != 'lin'):
170 
171  raise ValueError(Fore.YELLOW + "Non-lin observables specified. "
172  "Don't know how to handle that.")
173 
174  # Replace observables in assignment
175  import sympy as sp
176  for observable_id, formula in sigmas.items():
177  repl = sp.sympify(formula).subs(observable_id, observables[observable_id]['formula'])
178  sigmas[observable_id] = str(repl)
179 
180  if verbose:
181  print('Observables', len(observables))
182  print('Sigmas', len(sigmas))
183 
184  if not len(sigmas) == len(observables):
185  raise AssertionError(
186  f'Number of provided observables ({len(observables)}) and sigmas '
187  f'({len(sigmas)}) do not match.')
188 
189  fixed_parameters = get_fixed_parameters(condition_file, sbml_model)
190 
191  if verbose:
192  print("Overall fixed parameters", len(fixed_parameters))
193  print("Non-constant global parameters",
194  len(sbml_model.getListOfParameters()) - len(fixed_parameters))
195 
196  # Create Python module from SBML model
197  start = time.time()
198  sbml_importer.sbml2amici(
199  model_name,
200  output_dir=model_output_dir,
201  observables=observables,
202  constantParameters=fixed_parameters,
203  sigmas=sigmas,
204  allow_reinit_fixpar_initcond=allow_reinit_fixpar_initcond,
205  **kwargs)
206  end = time. time()
207 
208  if verbose:
209  print(f"{Fore.GREEN}Model imported successfully in "
210  f"{round(end - start, 2)}s")
